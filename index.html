    <script>

        const dapp = new MultiversxDapp({

            apiUrl: "https://gateway.multiversx.com", // URL de l'API MultiversX

        });


        const connectWalletButton = document.getElementById("connect-wallet");

        const walletAddressDisplay = document.getElementById("wallet-address");

        const incrementButton = document.getElementById("increment");

        const counterDisplay = document.getElementById("counter-display");


        let walletAddress = "";


        connectWalletButton.addEventListener("click", async () => {

            try {

                const address = await dapp.connectWallet();

                walletAddress = address;

                walletAddressDisplay.textContent = `Adresse : ${walletAddress}`;

            } catch (error) {

                console.error("Erreur lors de la connexion :", error);

            }

        });


        incrementButton.addEventListener("click", async () => {

            if (!walletAddress) {

                alert("Veuillez connecter votre portefeuille d'abord.");

                return;

            }


            try {

                // Appel simplifié d'une transaction sur le smart contract

                const result = await dapp.sendTransaction({

                    receiver: "adresse-du-smart-contract", // Adresse du contrat à remplacer

                    value: "0", // Aucun EGLD envoyé

                    data: "increment" // Fonction appelée dans le smart contract

                });


                console.log("Transaction envoyée :", result);

                alert("Compteur incrémenté avec succès !");


                // Simule la mise à jour du compteur

                const currentValue = parseInt(counterDisplay.textContent, 10);

                counterDisplay.textContent = currentValue + 1;

            } catch (error) {

                console.error("Erreur lors de l'incrémentation :", error);

            }

        });

    </script>
